----------------

2-run::: docker run --name mynode -dp 80:80 entty/cloud_project:v4

--- useful commands ---
docker rmi <image>

docker kill <container>
docker rm <container>

sudo docker build -t <name> <path-to-dockerfile>

docker run --name <containername> -dp <port1>:<port2> <imagename>

docker exec -ti <container> /bin/bash
(alt: docker attach?)

docker tag <oldname> <newname>

sudo docker push <tag>
